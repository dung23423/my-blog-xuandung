Giới Thiệu #

Khi mới học JavaScript, tôi thường bị “callback hell” hành hạ. Promises và async/await đã cứu rỗi cuộc đời coding của tôi. Hôm nay tôi sẽ chia sẻ cách xử lý bất đồng bộ hiệu quả nhất.

Sự phát triển của asynchronous programming trong JavaScript

Hiểu Về Asynchronous Programming #

Tại Sao Cần Async? #

// Synchronous - Blocking
console.log("Start");

// Giả sử function này mất 3 giây
function slowOperation() {
    const start = Date.now();
    while (Date.now() - start < 3000) {} // Block 3 giây
    return "Done";
}

const result = slowOperation(); // UI freeze 3 giây!
console.log(result);
console.log("End");

// Asynchronous - Non-blocking
console.log("Start");

setTimeout(() => {
    console.log("Done"); // Chạy sau 3 giây
}, 3000);

console.log("End"); // Chạy ngay lập tức
// Output: Start -> End -> Done

Promises - Giải Pháp Đầu Tiên #

Promise Là Gì? #

Promise đại diện cho một giá trị có thể có trong tương lai (hoặc lỗi).

// Tạo Promise
const myPromise = new Promise((resolve, reject) => {
    // Async operation
    const success = true;
    
    if (success) {
        resolve("Operation successful!"); // Thành công
    } else {
        reject("Operation failed!"); // Thất bại
    }
});

// Sử dụng Promise
myPromise
    .then(result => {
        console.log(result); // "Operation successful!"
    })
    .catch(error => {
        console.error(error);
    })
    .finally(() => {
        console.log("Cleanup"); // Luôn chạy
    });

Promise States #

// 1. Pending - Đang chờ
const pending = new Promise((resolve) => {
    setTimeout(() => resolve("Done"), 1000);
});

// 2. Fulfilled - Thành công
const fulfilled = Promise.resolve("Success");

// 3. Rejected - Thất bại
const rejected = Promise.reject("Error");

// Check state (không thể check trực tiếp, chỉ qua then/catch)
pending.then(result => console.log("Fulfilled:", result));
rejected.catch(error => console.log("Rejected:", error));

Creating Promises #

// Example 1: Timeout wrapper
function delay(ms) {
    return new Promise(resolve => {
        setTimeout(resolve, ms);
    });
}

// Usage
console.log("Start");
delay(2000).then(() => console.log("After 2 seconds"));

// Example 2: Fetch wrapper with retry
function fetchWithRetry(url, retries = 3) {
    return new Promise((resolve, reject) => {
        const attempt = (n) => {
            fetch(url)
                .then(resolve)
                .catch(error => {
                    if (n === 1) {
                        reject(error);
                    } else {
                        console.log(`Retry ${retries - n + 1}`);
                        setTimeout(() => attempt(n - 1), 1000);
                    }
                });
        };
        attempt(retries);
    });
}

// Example 3: Callback to Promise
function readFilePromise(filename) {
    return new Promise((resolve, reject) => {
        fs.readFile(filename, 'utf8', (error, data) => {
            if (error) {
                reject(error);
            } else {
                resolve(data);
            }
        });
    });
}

Error Handling #

// Try/catch với async/await
async function loadUserData(userId) {
    try {
        const response = await fetch(`/api/users/${userId}`);
        
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        
        const user = await response.json();
        return user;
    } catch (error) {
        console.error("Failed to load user:", error);
        // Re-throw hoặc return default value
        throw error; // hoặc return null;
    }
}

// Multiple try/catch blocks
async function complexOperation() {
    let user, orders;
    
    try {
        user = await fetchUser();
    } catch (error) {
        console.error("User fetch failed:", error);
        user = getDefaultUser();
    }
    
    try {
        orders = await fetchOrders(user.id);
    } catch (error) {
        console.error("Orders fetch failed:", error);
        orders = [];
    }
    
    return { user, orders };

Sequential vs Parallel Execution #

// Sequential - Chạy tuần tự (SLOW)
async function sequential() {
    const user = await fetchUser(); // Đợi 2 giây
    const posts = await fetchPosts(); // Đợi 2 giây
    const comments = await fetchComments(); // Đợi 2 giây
    // Tổng: 6 giây
    return { user, posts, comments };
}

// Parallel - Chạy song song (FAST)
async function parallel() {
    const [user, posts, comments] = await Promise.all([
        fetchUser(), // Chạy cùng lúc
        fetchPosts(), // Chạy cùng lúc
        fetchComments() // Chạy cùng lúc
    ]);
    // Tổng: 2 giây (thời gian của request lâu nhất)
    return { user, posts, comments };
}

// Parallel với error handling riêng
async function parallelSafe() {
    const results = await Promise.allSettled([
        fetchUser(),
        fetchPosts(),
        fetchComments()
    ]);
    
    return {
        user: results[0].status === 'fulfilled' ? results[0].value : null,
        posts: results[1].status === 'fulfilled' ? results[1].value : [],
        comments: results[2].status === 'fulfilled' ? results[2].value : []
    };
}

Real-World Examples #

1. API Request với Retry Logic #

async function fetchWithRetry(url, options = {}, retries = 3) {
    for (let i = 0; i < retries; i++) {
        try {
            const response = await fetch(url, options);
            
            if (!response.ok) {
                throw new Error(`HTTP ${response.status}`);
            }
            
            return await response.json();
        } catch (error) {
            const isLastAttempt = i === retries - 1;
            
            if (isLastAttempt) {
                throw error;
            }
            
            console.log(`Attempt ${i + 1} failed, retrying...`);
            
            // Exponential backoff
            const delay = Math.pow(2, i) * 1000;
            await new Promise(resolve => setTimeout(resolve, delay));
        }
    }
}

// Usage
try {
    const data = await fetchWithRetry('/api/unstable-endpoint');
    console.log(data);
} catch (error) {
    console.error("All retries failed:", error);
}

2. Loading Multiple Pages (Pagination) #

async function fetchAllPages(baseUrl) {
    const allData = [];
    let page = 1;
    let hasMore = true;
    
    while (hasMore) {
        try {
            const response = await fetch(`${baseUrl}?page=${page}`);
            const data = await response.json();
            
            allData.push(...data.items);
            
            hasMore = data.hasMore;
            page++;
            
            console.log(`Loaded page ${page - 1}`);
        } catch (error) {
            console.error(`Failed to load page ${page}:`, error);
            break;
        }
    }
    
    return allData;
}

// Usage
const allUsers = await fetchAllPages('/api/users');
console.log(`Loaded ${allUsers.length} users`);

3. Parallel Requests với Limit #

// Tránh gửi quá nhiều requests cùng lúc
async function fetchInBatches(urls, batchSize = 3) {
    const results = [];
    
    for (let i = 0; i < urls.length; i += batchSize) {
        const batch = urls.slice(i, i + batchSize);
        
        console.log(`Processing batch ${Math.floor(i / batchSize) + 1}`);
        
        const batchResults = await Promise.all(
            batch.map(url => fetch(url).then(r => r.json()))
        );
        
        results.push(...batchResults);
    }
    
    return results;
}

// Usage
const urls = Array.from({ length: 20 }, (_, i) => `/api/item/${i}`);
const data = await fetchInBatches(urls, 5); // 5 requests at a time

4. Race Condition với Loading State #

class DataLoader {
    constructor() {
        this.cache = new Map();
        this.pending = new Map();
    }
    
    async load(key, fetcher) {
        // Return cached data
        if (this.cache.has(key)) {
            return this.cache.get(key);
        }
        
        // Return pending promise (prevent duplicate requests)
        if (this.pending.has(key)) {
            return this.pending.get(key);
        }
        
        // Create new request
        const promise = fetcher()
            .then(data => {
                this.cache.set(key, data);
                this.pending.delete(key);
                return data;
            })
            .catch(error => {
                this.pending.delete(key);
                throw error;
            });
        
        this.pending.set(key, promise);
        return promise;
    }
    
    clear(key) {
        this.cache.delete(key);
        this.pending.delete(key);
    }
}

// Usage
const loader = new DataLoader();

// Multiple calls chỉ tạo 1 request
const [data1, data2, data3] = await Promise.all([
    loader.load('user-1', () => fetch('/api/user/1').then(r => r.json())),
    loader.load('user-1', () => fetch('/api/user/1').then(r => r.json())),
    loader.load('user-1', () => fetch('/api/user/1').then(r => r.json()))
]);
// Chỉ 1 request thực sự được gửi!

5. Timeout Wrapper #

function withTimeout(promise, ms) {
    return Promise.race([
        promise,
        new Promise((_, reject) =>
            setTimeout(() => reject(new Error('Timeout')), ms)
        )
    ]);
}

// Usage
try {
    const data = await withTimeout(
        fetch('/api/slow-endpoint'),
        5000 // 5 second timeout
    );
    console.log(data);
} catch (error) {
    if (error.message === 'Timeout') {
        console.error('Request took too long!');
    } else {
        console.error('Request failed:', error);
    }
}

// Advanced version with cleanup
function withTimeout(promise, ms, onTimeout) {
    let timeoutId;
    
    const timeoutPromise = new Promise((_, reject) => {
        timeoutId = setTimeout(() => {
            if (onTimeout) onTimeout();
            reject(new Error('Timeout'));
        }, ms);
    });
    
    return Promise.race([
        promise.then(result => {
            clearTimeout(timeoutId);
            return result;
        }),
        timeoutPromise
    ]);
}

6. Progress Tracking #

async function downloadWithProgress(urls, onProgress) {
    let completed = 0;
    const total = urls.length;
    
    const promises = urls.map(async (url) => {
        try {
            const response = await fetch(url);
            const data = await response.json();
            
            completed++;
            onProgress({
                completed,
                total,
                percentage: Math.round((completed / total) * 100)
            });
            
            return data;
        } catch (error) {
            completed++;
            onProgress({
                completed,
                total,
                percentage: Math.round((completed / total) * 100),
                error: url
            });
            return null;
        }
    });
    
    return await Promise.all(promises);
}

// Usage
const urls = Array.from({ length: 10 }, (_, i) => `/api/item/${i}`);

const results = await downloadWithProgress(urls, (progress) => {
    console.log(`Progress: ${progress.percentage}% (${progress.completed}/${progress.total})`);
    if (progress.error) {
        console.error(`Failed: ${progress.error}`);
    }
});

Advanced Patterns #

1. Queue System #

class AsyncQueue {
    constructor(concurrency = 1) {
        this.concurrency = concurrency;
        this.running = 0;
        this.queue = [];
    }
    
    async add(fn) {
        return new Promise((resolve, reject) => {
            this.queue.push({ fn, resolve, reject });
            this.run();
        });
    }
    
    async run() {
        if (this.running >= this.concurrency || this.queue.length === 0) {
            return;
        }
        
        this.running++;
        const { fn, resolve, reject } = this.queue.shift();
        
        try {
            const result = await fn();
            resolve(result);
        } catch (error) {
            reject(error);
        } finally {
            this.running--;
            this.run();
        }
    }
}

// Usage
const queue = new AsyncQueue(3); // Max 3 concurrent requests

const tasks = Array.from({ length: 20 }, (_, i) => 
    () => fetch(`/api/item/${i}`).then(r => r.json())
);

const results = await Promise.all(
    tasks.map(task => queue.add(task))
);

2. Debounced Async Function #

function debounceAsync(fn, delay) {
    let timeoutId;
    let latestResolve;
    let latestReject;
    
    return function(...args) {
        return new Promise((resolve, reject) => {
            if (timeoutId) {
                clearTimeout(timeoutId);
                latestReject(new Error('Cancelled'));
            }
            
            latestResolve = resolve;
            latestReject = reject;
            
            timeoutId = setTimeout(async () => {
                try {
                    const result = await fn.apply(this, args);
                    resolve(result);
                } catch (error) {
                    reject(error);
                }
            }, delay);
        });
    };
}

// Usage - Search API
const searchAPI = debounceAsync(async (query) => {
    const response = await fetch(`/api/search?q=${query}`);
    return response.json();
}, 500);

// Chỉ call API sau khi user ngừng gõ 500ms
searchInput.addEventListener('input', async (e) => {
    try {
        const results = await searchAPI(e.target.value);
        displayResults(results);
    } catch (error) {
        if (error.message !== 'Cancelled') {
            console.error('Search failed:', error);
        }
    }
});

3. Retry với Exponential Backoff #

async function retryWithBackoff(
    fn,
    maxRetries = 5,
    baseDelay = 1000,
    maxDelay = 30000
) {
    let lastError;
    
    for (let i = 0; i < maxRetries; i++) {
        try {
            return await fn();
        } catch (error) {
            lastError = error;
            
            if (i === maxRetries - 1) {
                throw error;
            }
            
            // Exponential backoff with jitter
            const exponentialDelay = Math.min(
                baseDelay * Math.pow(2, i),
                maxDelay
            );
            const jitter = Math.random() * 1000;
            const delay = exponentialDelay + jitter;
            
            console.log(`Retry ${i + 1}/${maxRetries} after ${Math.round(delay)}ms`);
            await new Promise(resolve => setTimeout(resolve, delay));
        }
    }
    
    throw lastError;
}

// Usage
const data = await retryWithBackoff(
    () => fetch('/api/unstable').then(r => r.json()),
    5,    // max 5 retries
    1000, // start with 1 second
    10000 // max 10 seconds
);

Error Handling Best Practices #

1. Custom Error Classes #

class APIError extends Error {
    constructor(message, statusCode, response) {
        super(message);
        this.name = 'APIError';
        this.statusCode = statusCode;
        this.response = response;
    }
}

class NetworkError extends Error {
    constructor(message) {
        super(message);
        this.name = 'NetworkError';
    }
}

async function fetchAPI(url) {
    try {
        const response = await fetch(url);
        
        if (!response.ok) {
            const errorData = await response.json();
            throw new APIError(
                errorData.message || 'API Error',
                response.status,
                errorData
            );
        }
        
        return await response.json();
    } catch (error) {
        if (error instanceof TypeError) {
            throw new NetworkError('Network connection failed');
        }
        throw error;
    }
}

// Usage with specific error handling
try {
    const data = await fetchAPI('/api/users');
} catch (error) {
    if (error instanceof APIError) {
        if (error.statusCode === 401) {
            redirectToLogin();
        } else if (error.statusCode === 404) {
            showNotFound();
        } else {
            showError(error.message);
        }
    } else if (error instanceof NetworkError) {
        showOfflineMessage();
    } else {
        showGenericError();
    }
}

2. Error Boundary Pattern #

async function withErrorBoundary(asyncFn, errorHandler) {
    try {
        return await asyncFn();
    } catch (error) {
        if (errorHandler) {
            return errorHandler(error);
        }
        throw error;
    }
}

// Usage
const loadUserData = () => 
    withErrorBoundary(
        async () => {
            const user = await fetchUser();
            const posts = await fetchUserPosts(user.id);
            return { user, posts };
        },
        (error) => {
            console.error('Failed to load user data:', error);
            return { user: null, posts: [] }; // Fallback value
        }
    );

const data = await loadUserData();

Performance Optimization #

1. Memoization #

class AsyncMemoizer {
    constructor() {
        this.cache = new Map();
    }
    
    async memoize(key, fn, ttl = 60000) {
        const cached = this.cache.get(key);
        
        if (cached && Date.now() - cached.timestamp < ttl) {
            console.log(`Cache hit: ${key}`);
            return cached.value;
        }
        
        console.log(`Cache miss: ${key}`);
        const value = await fn();
        
        this.cache.set(key, {
            value,
            timestamp: Date.now()
        });
        
        return value;
    }
    
    clear(key) {
        if (key) {
            this.cache.delete(key);
        } else {
            this.cache.clear();
        }
    }
}

// Usage
const memoizer = new AsyncMemoizer();

async function getUser(id) {
    return memoizer.memoize(
        `user-${id}`,
        () => fetch(`/api/users/${id}`).then(r => r.json()),
        300000 // Cache for 5 minutes
    );
}

2. Cancellation với AbortController #

async function fetchWithCancel(url, signal) {
    try {
        const response = await fetch(url, { signal });
        return await response.json();
    } catch (error) {
        if (error.name === 'AbortError') {
            console.log('Request cancelled');
            return null;
        }
        throw error;
    }
}

// Usage
const controller = new AbortController();

// Start request
const dataPromise = fetchWithCancel('/api/data', controller.signal);

// Cancel after 5 seconds
setTimeout(() => {
    controller.abort();
    console.log('Cancelled request');
}, 5000);

const data = await dataPromise;

// Real-world: Cancel on component unmount
class SearchComponent {
    constructor() {
        this.abortController = null;
    }
    
    async search(query) {
        // Cancel previous request
        if (this.abortController) {
            this.abortController.abort();
        }
        
        this.abortController = new AbortController();
        
        try {
            const response = await fetch(
                `/api/search?q=${query}`,
                { signal: this.abortController.signal }
            );
            const results = await response.json();
            this.displayResults(results);
        } catch (error) {
            if (error.name !== 'AbortError') {
                console.error('Search failed:', error);
            }
        }
    }
    
    destroy() {
        if (this.abortController) {
            this.abortController.abort();
        }
    }
}

Common Mistakes và Cách Khắc Phục #

1. Forgetting await #

// SAI - Quên await
async function loadData() {
    const data = fetchData(); // Returns Promise, not data!
    console.log(data.name); // undefined hoặc error
}

// ĐÚNG
async function loadData() {
    const data = await fetchData();
    console.log(data.name); // Works!
}

2. Not Handling Errors #

// SAI - Không handle errors
async function loadUser() {
    const user = await fetchUser(); // Có thể throw error
    return user;
}

// ĐÚNG
async function loadUser() {
    try {
        const user = await fetchUser();
        return user;
    } catch (error) {
        console.error('Failed to load user:', error);
        return null; // hoặc throw
    }
}

3. Sequential khi nên Parallel #

// SAI - Chậm (6 giây)
async function loadAll() {
    const users = await fetchUsers(); // 2s
    const posts = await fetchPosts(); // 2s
    const comments = await fetchComments(); // 2s
    return { users, posts, comments };
}

// ĐÚNG - Nhanh (2 giây)
async function loadAll() {
    const [users, posts, comments] = await Promise.all([
        fetchUsers(),
        fetchPosts(),
        fetchComments()
    ]);
    return { users, posts, comments };
}

4. Async trong forEach #

// SAI - Không chờ async operations
async function processItems(items) {
    items.forEach(async (item) => {
        await processItem(item); // Không đợi!
    });
    console.log('Done'); // Chạy ngay lập tức
}

// ĐÚNG - Cách 1: for...of
async function processItems(items) {
    for (const item of items) {
        await processItem(item);
    }
    console.log('Done'); // Chạy sau khi tất cả xong
}

// ĐÚNG - Cách 2: Promise.all (parallel)
async function processItems(items) {
    await Promise.all(
        items.map(item => processItem(item))
    );
    console.log('Done');
}

5. Unhandled Promise Rejections #

// SAI - Promise rejection không được handle
async function getData() {
    throw new Error('Failed');
}

getData(); // UnhandledPromiseRejectionWarning!

// ĐÚNG - Cách 1: Catch
getData().catch(error => console.error(error));

// ĐÚNG - Cách 2: Try/catch với await
async function main() {
    try {
        await getData();
    } catch (error) {
        console.error(error);
    }
}

main();

// Global handler
process.on('unhandledRejection', (error) => {
    console.error('Unhandled rejection:', error);
});

Những lỗi thường gặp với async/await

Kết Luận #

Async/await và Promises đã thay đổi cách chúng ta viết JavaScript:

Key Takeaways: #

  1. Promises - Đại diện cho values trong tương lai
  2. Async/await - Syntax đường giống synchronous code
  3. Promise.all() - Chạy song song để tối ưu performance
  4. Error handling - Luôn handle errors với try/catch
  5. Cancellation - Sử dụng AbortController khi cần
  6. Testing - Test async code kỹ lưỡng

Best Practices: #

  • Ưu tiên async/await thay vì .then()
  • Sử dụng Promise.all() cho parallel operations
  • Luôn handle errors
  • Tránh mixing callbacks và Promises
  • Cancel requests khi không cần thiết
  • Cache kết quả khi có thể

Từ kinh nghiệm của tôi, việc hiểu rõ async programming không chỉ giúp code chạy nhanh hơn mà còn tránh được rất nhiều bugs khó debug.
